نمونه تست ACOC

@pytest.mark.parametrize("size", [[], [1 item], [2+]])
@pytest.mark.parametrize("order", ["sorted","reverse","random"])
@pytest.mark.parametrize("sort_field", ["title","min_level","time_limit"])
@pytest.mark.parametrize("sort_algo", ["bubble","merge"])
@pytest.mark.parametrize("query", ["","آزمون","ناموجود"])
@pytest.mark.parametrize("search_algo", ["linear","binary"])
def test_acoc_catalog_combinations(...):
    result = process_catalog(items, query=..., sort_algo=..., ...)
    assert result["meta"]["total_after"] == len(result["items"])
    assert result["meta"]["total_after"] <= result["meta"]["total_before"]
    assert result["meta"]["total_before"] == len(items)  # ← kill PC-01
